Aelter skript:
Nur herunterladen, klassifizieren, Parquet schreiben:

python roco_pipeline.py \
  --hf-token hf_xxx \
  --download-dir ./roco_local_parquet \
  --output-dir ./roco_outputs \
  --device cuda

Mit Bildexport:

python roco_pipeline.py \
  --hf-token hf_xxx \
  --download-dir ./roco_local_parquet \
  --output-dir ./roco_outputs \
  --device cuda \
  --export-images

Mit Heuristik-Fallback:

python roco_pipeline.py \
  --hf-token hf_xxx \
  --download-dir ./roco_local_parquet \
  --output-dir ./roco_outputs \
  --device cuda \
  --use-heuristics

Nur Train-Split:

python roco_pipeline.py \
  --hf-token hf_xxx \
  --splits train
